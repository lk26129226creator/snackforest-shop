💻 程式碼範例

基本用法

// 基本 API 呼叫 const result = await api.getProducts(); if (result.success) { console.log('商品資料:', result.data); } // 載入表格到容器 loadTable('/api/products', '#my-container'); // 新增商品 const addResult = await api.addProduct({ name: '新商品', price: 100, categoryId: 1 });

進階用法

// 自訂表格選項 loadTable('/api/products', '#container', { excludeColumns: ['created_at'], customColumns: { 'ProductName': '商品名稱' }, actions: [ { text: '編輯', className: 'btn-sm', onclick: 'editProduct({id})' } ] }); // 載入所有基本資料 const allData = await api.loadAllBasicData(); console.log('商品:', allData.products); console.log('分類:', allData.categories);
